"use strict";
const database_1 = require("../../services/database");
const mongoose_1 = require("mongoose");
const schema = new mongoose_1.Schema({
    age: {
        type: Number
    },
    name: {
        type: String,
        required: true
    },
    create: {
        type: Date,
        "default": Date.now
    },
    description: {
        type: String
    }
});
schema.static("updateAuthor", (author, description) => {
    return exports.Author
        .update({
        "_id": author
    }, {
        "$set": {
            "description": description
        }
    })
        .exec();
});
schema.static("updateByAge", (ageLimit, text) => {
    return exports.Author
        .where("age")
        .gte(ageLimit)
        .update({
        "$set": {
            description: text
        }
    })
        .exec();
});
exports.Author = database_1.mongoose.model("Author", schema);
